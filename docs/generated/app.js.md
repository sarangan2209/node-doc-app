/**
 * Greets a user by name.
 * @param {string} name - The name of the user.
 * @returns {string} A greeting message.
 * @example
 * // returns "Hello-test, Alice!"
 * greet("Alice");
 */

function greet(name) {
    return `Hello test, ${name}!`;
}

console.log(greet("World")); // Output: "Hello test, World!"
